import { motion } from 'framer-motion';
import { Shirt, Footprints, Gem, Scissors, Lightbulb, Palette } from 'lucide-react';

const OutfitRow = ({ icon: Icon, label, value }) => {
  if (!value) return null;
  return (
    <div className="flex items-start gap-3 py-3 border-b border-border/40 last:border-0">
      <div className="mt-0.5 p-2 rounded-lg bg-secondary text-accent">
        <Icon className="w-4 h-4" />
      </div>
      <div>
        <p className="text-xs font-medium text-muted-foreground uppercase tracking-wider mb-0.5">{label}</p>
        <p className="text-sm text-foreground leading-relaxed">{value}</p>
      </div>
    </div>
  );
};

const VisualCard = ({ emoji, label, value }) => {
  if (!value) return null;
  return (
    <div className="bg-card rounded-xl border border-border/50 p-4 flex flex-col items-center text-center gap-2">
      <div className="text-3xl">{emoji}</div>
      <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider">{label}</p>
      <p className="text-sm text-foreground leading-relaxed">{value}</p>
    </div>
  );
};
